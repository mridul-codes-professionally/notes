cd .. = go to previous dir

dir *(file extention) = show files with that extention
dir /a = show all files including hidden

mkdir (name) = make new directories 
rmdir (name) = delete folder
rmdir /s (name) = delete a directory which has files in it with files
echo > (filename+extention) = create a new file  (file has written echo is on inside it )
del (filename+extention) = delete file
type null >(filename+extention) = make a new empty file
type (filename+extention) = print file content
